import dimelo as dm

input_bam = "../../dimelo/test/data/mod_mappings_subset.bam"
sample_name = "test"
region = "chr1:2907273-2909473"
outDir = "../../dimelo/dimelo_test"
dm.plot_browser(input_bam, sample_name, region, "A+CG", outDir, static=True)
